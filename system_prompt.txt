### ESQUEMA DE LA BASE DE DATOS

-- Tablas de Dimension: Ubicaciones Geográficas
CREATE TABLE Continent(
    ContinentKey int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    ContinentName varchar(20) NOT NULL -- Nombre del continente
);

-- Tablas de Dimension: Ubicaciones por Paises
CREATE TABLE Country(
    CountryKey int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    CountryName varchar(40) NOT NULL, -- Nombre del país
    CountryCode varchar(3) NOT NULL, -- Código de país
    CountryCapital varchar(40) NOT NULL, -- Capital
    Population int NULL, -- Población
    Subdivision text NULL, -- Subdivisión política
    ContinentKey int NOT NULL -- Relación con Continente
);

-- Tablas de Dimension: Ubicaciones por Estados
CREATE TABLE State(
    StateKey int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    StateName varchar(40) NOT NULL, -- Nombre del estado/provincia
    EnglishStateName varchar(40) NULL, -- Nombre en inglés
    StateType varchar(40) NOT NULL, -- Tipo de estado
    StateCode varchar(3) NULL, -- Código de estado
    StateCapital varchar(40) NULL, -- Capital del estado
    RegionName varchar(40) NULL, -- Región interna
    RegionCode varchar(3) NULL, -- Código de región
    CountryKey int NOT NULL -- Relación con País
);

-- Tablas de Dimension: Ubicaciones por Ciudades
CREATE TABLE City(
    CityKey int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    CityName varchar(40) NOT NULL, -- Nombre de la ciudad
    StateKey int NOT NULL -- Relación con Estado
);

-- Tablas de Dimension: Datos de los proveedores comerciales
CREATE TABLE Supplier(
    SupplierKey int PRIMARY KEY, -- Clave del proveedor
    CompanyName varchar(40) NOT NULL, -- Nombre de la empresa proveedora
    Address varchar(60) NULL, -- Dirección
    PostalCode varchar(10) NULL, -- Código postal
    CityKey int NULL -- Relación con Ciudad
);

-- Tablas de Dimension: Datos de los consumidores/clientes
CREATE TABLE Customer(
    CustomerKey int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, -- Clave del cliente
    CustomerID char(5) NOT NULL, -- ID transaccional del cliente
    CompanyName varchar(40) NOT NULL, -- Nombre de la empresa cliente
    Address varchar(60) NULL, -- Dirección
    PostalCode varchar(10) NULL, -- Código postal
    CityKey int NULL -- Relación con Ciudad
);

-- Tablas de Dimension: Datos de los empleados
CREATE TABLE Employee(
    EmployeeKey int PRIMARY KEY, -- Clave del empleado
    FirstName varchar(10) NOT NULL, -- Nombre
    LastName varchar(20) NOT NULL, -- Apellido
    Title varchar(30) NULL, -- Cargo
    TitleOfCourtesy varchar(25) NULL, -- Título de cortesía
    BirthDate date NULL, -- Fecha de nacimiento
    HireDate date NULL, -- Fecha de contratación
    Address varchar(60) NULL, -- Dirección residencial
    City varchar(15) NULL, -- Ciudad (texto)
    Region varchar(15) NULL, -- Región (texto)
    PostalCode varchar(10) NULL, -- Código postal
    Country varchar(15) NULL, -- País (texto)
    SupervisorKey int NULL -- Clave del jefe directo
);

-- Tablas de Dimension: Relacion de los territorios en donde trabajan/dependen los empleados
CREATE TABLE Territories(
    EmployeeKey int NOT NULL, -- Relación con Empleado
    CityKey int NOT NULL, -- Relación con Ciudad
    CONSTRAINT PK_Territories_EmployeeKey_CityKey PRIMARY KEY (EmployeeKey, CityKey)
);

-- Tablas de Dimension: Categorias de producto
CREATE TABLE Category(
    CategoryKey int PRIMARY KEY, -- Clave de la categoría
    CategoryName varchar(15) NOT NULL, -- Nombre de la categoría
    Description text NULL -- Descripción de categoría
);

- Tablas de Dimension: Catalogo de productos
CREATE TABLE Product(
    ProductKey int PRIMARY KEY, -- Clave del producto
    ProductName varchar(40) NOT NULL, -- Nombre del producto
    QuantityPerUnit varchar(20) NULL, -- Cantidad por unidad
    UnitPrice money NULL, -- Precio unitario base
    Discontinued bit NOT NULL, -- Descontinuado ('0'=No, '1'=Sí)
    CategoryKey int NULL -- Relación con Categoría
);

-- Tablas de Dimension: Datos de los transportistas
CREATE TABLE Shipper(
    ShipperKey int PRIMARY KEY, -- Clave del transportista
    CompanyName varchar(40) NOT NULL -- Nombre de la empresa de transporte
);

-- Tablas de Dimension: Tiempo (Calendario)
CREATE TABLE Date(
    DateKey int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, -- Clave de fecha (AAAAMMDD)
    Date date NULL, -- Fecha formato date
    DayNbWeek smallint NULL, -- Número de día de la semana (1-7)
    DayNameWeek varchar(10) NULL, -- Nombre del día
    DayNbMonth smallint NULL, -- Día del mes (1-31)
    DayNbYear smallint NULL, -- Día del año (1-366)
    WeekNbYear smallint NULL, -- Semana del año (1-53)
    MonthNumber smallint NULL, -- Número de mes (1-12)
    MonthName varchar(10) NULL, -- Nombre del mes
    Quarter smallint NULL, -- Trimestre (1-4)
    Semester smallint NULL, -- Semestre (1-2)
    Year int NULL -- Año
);

-- Tabla de Hechos: Ventas por detalle
CREATE TABLE Sales(
    CustomerKey int NOT NULL, -- Relación con Cliente
    EmployeeKey int NOT NULL, -- Relación con Empleado
    OrderDateKey int NOT NULL, -- Relación con Fecha de Orden
    DueDateKey int NOT NULL, -- Relación con Fecha de Vencimiento
    ShippedDateKey int NOT NULL, -- Relación con Fecha de Envío
    ShipperKey int NOT NULL, -- Relación con Transportista
    ProductKey int NOT NULL, -- Relación con Producto
    SupplierKey int NULL, -- Relación con Proveedor
    OrderNo int NOT NULL, -- Número de Pedido
    OrderLineNo int NOT NULL, -- Línea de Detalle del Pedido
    UnitPrice money NOT NULL, -- Precio unitario vendido
    Quantity smallint NOT NULL, -- Cantidad vendida
    Discount real NOT NULL, -- Descuento aplicado
    SalesAmount money NOT NULL, -- Monto total de la venta (Métrica)
    Freight money NOT NULL -- Costo de flete
);
